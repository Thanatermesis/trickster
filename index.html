class NewPersonEvent
  MAX_TRIES = 5
  def self.perform(id)
    tried = 0
    person = Person.find_by_id(id)
    while person.nil? && tried < MAX_TRIES
      sleep(tried)
      tried += 1
      person = Person.find_by_id(id)
    end
    if person.nil?
      raise "No such person with id #{id}"
    else
      UserMailer.send_welcome_email(person)
    end
  end
end

  end

# After
class MailerServiceController
  def create
    type       = params[:mail_type]
    data       = params[:data]
    request_id = params[:request_id]

    if request = Request.find_by_request_id(request_id)
      respond_with(request.results)
    else
      results = Mailers.find(type).mail(data)
      respond_with(Request.create!(:request_id => request_id,
                                   :results    => results).results)
    end
  end
end

  person = Person.find_by_id(id)

What can you do?

Don't over-engineer

1 - Historical Record

  • Log, log, log
  • Sanity Check invariants
  • Audit Activity

2 - Prevent Bad Data

  • Database constraints
  • AR Validations cannot be trusted
  • Fix bad data

3 - Fix Errors

  • Email redalert@yourcompany.com
  • Fix the problem
  • Prevent the error
  • Log warnings

4 - Monitor

  • Establish baseline for "normal"
  • Alert outside of this baseline
  • Refine

New Services

  • Prevent client "abuse"
  • Design for Idempotence

Stop scratching your head

  • Record history
  • Prevent bad data
  • Fix errors
  • Monitor
  • Design for Disaster
